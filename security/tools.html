# 名前解決しない：-n

ssh

リモートホストにコピー

cat file1 | ssh -f user@server 'cd $HOME; cat > file1'

踏み台設定

cat ~/.ssh/config
Host host2
     ProxyCommand ssh host1 nc %h %p
